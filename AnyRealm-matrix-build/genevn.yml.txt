name: Build Custom Kernel - Genevn

on:
  push:
    branches: [ main ]

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Code
        uses: actions/checkout@v4
        sparse-checkout: ["device-profiles.json"]
      - name: Initialize Full Build Environment
        run: |
           Install the complete suite of build tools and libraries
          sudo apt-get update
          sudo apt-get install build-essential libncurses5-dev libncursesw5-dev \
            libqt5core5a libqt5gui5 libqt5widgets5 qtbase5-dev \
            clang lld llvm gcc-aarch64-linux-gnu binutils bc bison flex \
            libssl-dev libelf-dev cpio kmod python3 -y \
          
          # Clone the repository
          git clone --depth=1 https://github.com/015stray91/kernel-msm-1.git kernel-source
          
          # Navigate into the kernel directory
          cd kernel-source
          
          # Force a clean state to prevent 'make' from looking at parent directories
          make mrproper
          
          # Ensure output directory exists locally
          mkdir -p out
          
          # Merge configurations
          ./scripts/kconfig/merge_config.sh  \
            arch/arm64/configs/genevn.defconfig \
            arch/arm64/configs/nethunter_defconfig \
            arch/arm64/configs/vendor/parrot_consolidate.config \
            arch/arm64/configs/vendor/parrot_GKI.config \
            arch/arm64/configs/vendor/ext_config/moto-parrot.config \
            arch/arm64/configs/vendor/ext_config/moto-parrot-gki.config \
            build.config.msm.parrot \
            arch/arm64/configs/vendor/ext_config/moto-parrot-genevn.config \
            
          # Copy the merged config to the out directory to satisfy the build system
          cp .config out/
          
          # Finalize config using the local directory
          make ARCH=arm64 O=out msm_defconfig

      - name: Build Kernel
        run: |
          cd kernel-source
          # Use all available CPU cores for the build
          make -j$(nproc) ARCH=arm64 O=out
